Q2. Write a Java program that works as a simple calculator. Use a grid layout to arrange buttons for the digits and for the +, -,*, % operations. Add a text field to display the result. Handle any possible exceptions like divided by zero.

Aim: To create a java program that works as a simple calculator. Use a grid layout to arrange buttons for the digits and for the +, -,*, % operations. Add a text field to display the result. Handle any possible exceptions like divided by zero.

System and Software Requirements: A Computer with any Operating system with javac compiler installed in it.

Description:
A Java program that works as a simple calculator can be designed using a grid layout to arrange buttons for the digits and for the +, -, *, and % operations, and a text field to display the result of calculations. The program should be able to handle any possible exceptions that may arise during calculations, such as attempts to divide by zero.
To implement this program, the following theoretical explanation can be provided:
1. GUI Design: The program should use a grid layout to arrange buttons for the digits 0-9 and for the +, -, *, and % operations. Additionally, a text field should be included to display the result of calculations. The GUI should be designed to be user-friendly and intuitive.
2. User Input: The program should be designed to handle user inputs, such as button clicks, and perform calculations based on those inputs. When a user clicks a button, the program should detect which button was clicked and perform the corresponding arithmetic operation based on the current state of the calculator.
3. Calculation Logic: The program should implement the necessary code to perform arithmetic operations (addition, subtraction, multiplication, and modulus) based on user inputs. The program should keep track of the current state of the calculator, such as the first and second operands, and update the text field with the result of each calculation.
4. Error Handling: The program should be designed to handle any possible exceptions that may arise during use, such as attempts to divide by zero. In such cases, the program should display an appropriate error message to the user and reset the calculator to its default state.
5. Testing: The program should be tested thoroughly to ensure that it functions correctly and handles exceptions appropriately. Testing should include edge cases and potential errors to ensure that the program is reliable and free of bugs.
In summary, a Java program that works as a simple calculator can be designed using a grid layout to arrange buttons for the digits and arithmetic operators, and a text field to display the result of calculations. The program should be designed to handle any possible exceptions, such as attempts to divide by zero, and should be tested thoroughly to ensure reliability and accuracy.

import javax.swing.*;
import java.awt.*;
import java.awt.event.*;
public class Calculator extends JFrame implements ActionListener {
// Declare buttons and text field
private JButton buttons[];
private JTextField textField;
// Declare variables for calculation
private double firstOperand;
private double secondOperand;
private String operator;
public Calculator() {
// Set up the frame and layout
super("Calculator");
setSize(300, 300);
setResizable(false);
setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);
setLayout(new GridLayout(4, 4));
// Initialize buttons
buttons = new JButton[16];
for (int i = 0; i < 10; i++) {
buttons[i] = new JButton(String.valueOf(i));
buttons[i].addActionListener(this);
add(buttons[i]);
}
buttons[10] = new JButton("+");
buttons[11] = new JButton("-");
buttons[12] = new JButton("*");
buttons[13] = new JButton("/");
buttons[14] = new JButton("%");
buttons[15] = new JButton("=");
for (int i = 10; i < 16; i++) {
buttons[i].addActionListener(this);
add(buttons[i]);
}
// Initialize text field
textField = new JTextField();
textField.setEditable(false);
add(textField);
// Set initial state
firstOperand = 0;
secondOperand = 0;
operator = "";
}
public void actionPerformed(ActionEvent e) {
// Get the button text
String button = e.getActionCommand();
// If a digit is pressed
if (button.matches("[0-9]")) {
if (operator.equals("")) {
firstOperand = firstOperand * 10 + Double.parseDouble(button);
textField.setText(String.valueOf(firstOperand));
} else {
secondOperand = secondOperand * 10 + Double.parseDouble(button);
textField.setText(String.valueOf(secondOperand));
}
}
// If an operator is pressed
else if (button.equals("+") || button.equals("-") || button.equals("*") || button.equals("/") || button.equals("%")) {
operator = button;
}
// If equals is pressed
else if (button.equals("=")) {
double result = 0;
try {
switch (operator) {
case "+":
result = firstOperand + secondOperand;
break;
case "-":
result = firstOperand - secondOperand;
break;
case "*":
result = firstOperand * secondOperand;
break;
case "/":
if (secondOperand == 0) {
throw new ArithmeticException("Division by zero!");
}
result = firstOperand / secondOperand;
break;
case "%":
if (secondOperand == 0) {
throw new ArithmeticException("Modulo by zero!");
}
result = firstOperand % secondOperand;
break;
}
textField.setText(String.valueOf(result));
} catch (ArithmeticException ex) {
textField.setText(ex.getMessage());
}
firstOperand = result;
secondOperand = 0;
operator = "";
}
}
public static void main(String[] args) {
Calculator calculator = new Calculator();
calculator.setVisible(true);
}
}

import javax.swing.*;
import java.awt.*;
import java.awt.event.*;
public class Calculator extends JFrame implements ActionListener {
private JTextField display;
private double currentValue = 0.0;
private String currentOperation = "";
private boolean clearOnNextDigit = false;
public Calculator() {
super("Calculator");
JPanel panel = new JPanel(new GridLayout(4, 4));
display = new JTextField();
display.setEditable(false);
JButton[] buttons = new JButton[16];
String[] buttonLabels = {"7", "8", "9", "/", "4", "5", "6", "*", "1", "2", "3", "-", "0", ".", "%", "+"};
for (int i = 0; i < 16; i++) {
buttons[i] = new JButton(buttonLabels[i]);
buttons[i].addActionListener(this);
panel.add(buttons[i]);
}
setLayout(new BorderLayout());
add(display, BorderLayout.NORTH);
add(panel, BorderLayout.CENTER);
setSize(300, 300);
setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);
setResizable(false);
}
public void actionPerformed(ActionEvent e) {
String command = e.getActionCommand();
try {
double value = Double.parseDouble(command);
if (clearOnNextDigit) {
currentValue = 0.0;
clearOnNextDigit = false;
}
currentValue = currentValue * 10 + value;
display.setText(Double.toString(currentValue));
} catch (NumberFormatException ex) {
if (command.equals(".")) {
if (!display.getText().contains(".")) {
display.setText(display.getText() + ".");
}
} else if (command.equals("+") || command.equals("-") || command.equals("*") || command.equals("/") || command.equals("%")) {
if (currentOperation.equals("")) {
currentValue = Double.parseDouble(display.getText());
clearOnNextDigit = true;
} else {
performOperation();
}
currentOperation = command;
} else if (command.equals("=")) {
performOperation();
currentOperation = "";
}
}
}
private void performOperation() {
double newValue = Double.parseDouble(display.getText());
double result = 0.0;
try {
switch (currentOperation) {
case "+":
result = currentValue + newValue;
break;
case "-":
result = currentValue - newValue;
break;
case "*":
result = currentValue * newValue;
break;
case "/":
if (newValue == 0) {
throw new ArithmeticException("Divide by zero");
}
result = currentValue / newValue;
break;
case "%":
if (newValue == 0) {
throw new ArithmeticException("Modulo by zero");
}
result = currentValue % newValue;
break;
}
currentValue = result;
display.setText(Double.toString(result));
clearOnNextDigit = true;
} catch (ArithmeticException ex) {
display.setText(ex.getMessage());
clearOnNextDigit = true;
}
}
public static void main(String[] args) {
Calculator calculator = new Calculator();
calculator.setVisible(true);
}
}
