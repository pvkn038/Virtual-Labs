Q8. Suppose that a table named Table.txt is stored in a text file. The first line in the file is the header, and the remaining lines correspond to rows in the table. The elements are separated by commas. Write a java program to display the table using Labels in GridLayout.

Aim: To develop a Java program that can read a text file containing a table, where the first line represents the header and the remaining lines correspond to rows in the table, and display the table using Labels in GridLayout.

System and Software Requirements: A Computer with any Operating system with javac compiler installed in it.

import java.awt.GridLayout;
import java.io.BufferedReader;
import java.io.FileReader;
import java.io.IOException;
import javax.swing.JFrame;
import javax.swing.JLabel;
public class TableDisplay {
public static void main(String[] args) {
JFrame frame = new JFrame("Table Display");
frame.setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);
try {
BufferedReader reader = new BufferedReader(new FileReader("Table.txt"));
// Get the headers
String headerLine = reader.readLine();
String[] headers = headerLine.split(",");
// Get the number of rows and columns
int numRows = 0;
while (reader.readLine() != null) {
numRows++;
}
int numCols = headers.length;
// Create the labels for the table
JLabel[][] labels = new JLabel[numRows + 1][numCols];
for (int col = 0; col < numCols; col++) {
labels[0][col] = new JLabel(headers[col]);
}
// Read the data and create the labels for the rows
reader = new BufferedReader(new FileReader("Table.txt"));
reader.readLine(); // Skip the header line
String line;
int row = 1;
while ((line = reader.readLine()) != null) {
String[] values = line.split(",");
for (int col = 0; col < numCols; col++) {
labels[row][col] = new JLabel(values[col]);
}
row++;
}
reader.close();
// Add the labels to the frame using a GridLayout
frame.setLayout(new GridLayout(numRows + 1, numCols));
for (int rowIdx = 0; rowIdx <= numRows; rowIdx++) {
for (int colIdx = 0; colIdx < numCols; colIdx++) {
frame.add(labels[rowIdx][colIdx]);
}
}
frame.pack();
frame.setVisible(true);
} catch (IOException e) {
System.err.println("Error: " + e.getMessage());
}
}
}
